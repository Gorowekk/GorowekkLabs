#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int M, N;
    
    // ввод размеров
    cout << "размер массива A: ";
    cin >> M;
    cout << "размер массива B: ";
    cin >> N;

    // создание массивов A и B
    vector<int> A(M);
    vector<int> B(N);

    // ввод элементов массива A
    cout << "элементы массива A:" << endl;
    for (int i = 0; i < M; ++i) {
        cin >> A[i];
    }

    // ввод элементов массива B
    cout << "элементы массива B:" << endl;
    for (int i = 0; i < N; ++i) {
        cin >> B[i];
    }

    // сортировка массивов A и B
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // массив для хранения 
    vector<int> C;

    // set_difference для нахождения элементов
    set_difference(A.begin(), A.end(), B.begin(), B.end(), back_inserter(C));

    
    cout << "элементы из массива A, которых нет в массиве B:" << endl;
    for (int num : C) {
        cout << num << " ";
    }
    cout << endl;

    return 0;
}
