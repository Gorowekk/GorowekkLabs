#include <iostream>
#include <vector>
using namespace std;

// функция для вставки блока элементов в массив, начиная с указанного индекса
void insertBlock(vector<int>& array, const vector<int>& block, int index) {
    if (index >= 0 && index <= array.size()) {
        array.insert(array.begin() + index, block.begin(), block.end());
    } else {
        cout << "ошибка: недопустимый индекс для вставки." << endl;
    }
}

int main() {
    vector<int> a = {1, 2, 3};
    vector<int> b = {4, 5, 6};

    // вывод исходного массива
    cout << "исходный массив: ";
    for (int num : a) {
        cout << num << " ";
    }
    cout << endl;

    // вставка блока элементов начиная с индекса 1
    insertBlock(a, b, 1);

    cout << "массив после вставки блока: ";
    for (int num : a) {
        cout << num << " ";
    }
    cout << endl;

    return 0;
}
