#include <iostream>
#include <vector>

using namespace std;

// функция для удаления блока элементов из массива, начиная с указанного индекса
void removeBlock(vector<int>& array, int index, int blockSize) {
    if (index >= 0 && index < array.size() && blockSize > 0) {
        array.erase(array.begin() + index, array.begin() + index + blockSize);
    }
}

int main() {
    vector<int> a = {1, 2, 3, 4, 5, 6};

    // исходный массив
    cout << "исходный массив: ";
    for (int num : a) {
        cout << num << " ";
    }
    cout << endl;

    // удаление блока элементов начиная с индекса 2 и размером 3
    removeBlock(a, 2, 3);

    // массив после удаления блока
    cout << "массив после удаления блока: ";
    for (int num : a) {
        cout << num << " ";
    }
    cout << endl;

    return 0;
}
